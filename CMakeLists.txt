cmake_minimum_required(VERSION 3.7.2)

project(sel4-riscv-vmm C)

set(configure_string "")

file(GLOB static
    src/*.c src/tests/*.c src/arch/${KernelArch}/*.c src/plat/${KernelPlatform}/*.c
    src/arch/${KernelArch}/*.S src/plat/${KernelPlatform}/*.S
)

MakeCPIO(archive.o "${CMAKE_SOURCE_DIR}/projects/sel4_riscv_vmm/linux/linux;${CMAKE_SOURCE_DIR}/projects/sel4_riscv_vmm/linux/linux-dtb")

add_executable(sel4-riscv-vmm EXCLUDE_FROM_ALL ${static} archive.o)
target_link_libraries(sel4-riscv-vmm Configuration muslc sel4 sel4runtime sel4allocman sel4vka sel4utils sel4platsupport sel4muslcsys)
target_compile_options(sel4-riscv-vmm PRIVATE -Werror -g)

# Set this image as the rootserver
DeclareRootserver(sel4-riscv-vmm)
GenerateSimulateScript()
